c语言实现单词达人

字数统计: 2.5k | 阅读时长 ≈ 11

例2为单词小助手，要求完成以下任务：
1.改写程序为良好程序风格（文档注释，函数注释，语句注释）。
2.将单词测试中的功能完善，可针对做错的单词重复记忆。
3.查询单词的功能添加英文词查询，中文查询的功能完善（考虑如何显示同样中文意义，不同的英文单词）

代码 

#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define MAX_CHAR  20  // 最大字符
#define MAX_NUM  200  // 单词的最大个数

 //单词的结构体
 struct word
{
   char  en[MAX_CHAR]; // 英文形式
   char  ch[MAX_CHAR];   //中文形式
}  s[MAX_NUM]; //单词数组
int   num;           //单词个数
int select=1;//select 为是否退出系统的标记
int d=0,c=0;

//帮助
void help()
{
    printf("\n欢迎来到单词达人");
    printf("\n本系统主要实现英语单词学习的功能");
    printf("\n词库维护0：用户可对词典文件中的单词进行预览，增删改查。并且中文查询提供显示同样中文意义，不同的英文单词的服务");
    printf("\n单词预览1：将词库中的所有单词显示。");
    printf("\n单词背诵（中译英）2：用户需要根据随机输出的中文写出正确的英语拼写，系统会根据用户输入情况，对输入错误的单词加强记忆。"); 
    printf("\n单词背诵（英译中）3：用户需要根据随机输出的英文写出正确的中文拼写，系统会根据用户输入情况，对输入错误的单词加强记忆。");
    printf("\n查询成绩4：用户在测试的过程中，输入正确的单词加10分，输入错误的单词扣10分，本功能可以提供用户正确和错误的次数以及最后的总得分。");
    printf("\n帮助5：查看本系统的所有功能服务，方便用户使用。");
    printf("\n退出系统6：输入6即可退出，欢迎再次使用。"); 
}

//从文件中读取单词的信息
void  readfile()
{
   FILE *fp;
   int i=0;
   fp=fopen("data.txt","r");//只读打开文件 
   if(!fp)
   {
       printf("\n打开文件data.txt失败!");

   }
   while(fscanf(fp,"%s %s ",s[i].en,s[i].ch)==2)
   {
       i++;
   }
   num=i;
   if(0==i)
       printf("\n文件为空，请选择词典维护增加词条！");
   else
       printf("\n");
   fclose(fp);

}

//从文件中写单词的信息
void  writefile()
{
   FILE *fp;
   int i=0;
   fp=fopen("data.txt","w");//只写打开文件 
   if(!fp)
   {
       printf("\n打开文件data.txt失败!");

   }
   for(i=0;i<num;i++)
   {
       fprintf(fp,"\n%s %s ",s[i].en,s[i].ch);//在文件种打印单词的英语及汉语 
   }
   printf("\n");
   fclose(fp);
}

//按字典排序
void sort()
{
    int i,j;
    char temp[MAX_CHAR];
    for(i=0;i<num-1;i++)
    {
        for(j=num-1;j>i;j--)
            if(strcmp(s[j-1].en,s[j].en)>0)//比较最后一个单词与倒数第二个单词，若str1=str2，则返回零；若str1<str2，则返回负数；若str1>str2，则返回正数
            {    
                //当倒数第二个单词的字符串大于最后一个单词的字符串，将2个位置交换 
                strcpy(temp,s[j-1].en);//复制，将倒数第二个单词复制给temp 
                strcpy(s[j-1].en,s[j].en);//复制，将倒数第一个单词复制给倒数第二个 
                strcpy(s[j].en,temp);//复制，将temp复制给倒数第一个单词 
                strcpy(temp,s[j-1].ch);
                strcpy(s[j-1].ch,s[j].ch);
                strcpy(s[j].ch,temp);

            }
    }
}

//添加单词信息
void add()
{
   int i=num,j,flag=1; 
   while(flag) //flag为1作为添加单词的标志 
    {
        flag=0;
        printf("\n请输入单词的英文形式:");
        scanf("%s",s[i].en);
        for(j=0;j<i;j++)

            //检验是否已有这个单词 
            if(strcmp(s[i].en,s[j].en)==0)
            {
                printf("已有该单词,请检查后重新录入!\n");
                flag=1;
                break; /*如有重复立即退出该层循环,提高判断速度*/
            }

    }
   printf("\n请输入单词的中文形式:");
   scanf("%s",s[i].ch);
   num++;
   printf("\n您输入的信息为: 英文: %s 中文: %s  ",s[i].en,s[i].ch);
    sort();//调用字典排序 
}

//删除单词信息
void del()
{
    int i=0,j=0;
    char  en[MAX_CHAR];   //英文形式
      printf("\n请输入你要删除的单词英文形式:");    
    scanf("%s",en);
    for(i=0;i<num;i++)//先找到该英文形式对应的序号
        if(strcmp(s[i].en,en)==0)//找到时删除该单词 
        {
            for(j=i;j<num-1;j++)
                 s[j]=s[j+1];
            num--; //数量减少 1
            return;
        }
    printf("\n没有这个单词!");

}

//修改单词信息
void modify()
{
    int i=0,choose=0,flag=1;//chooses代表选项标识，flag代表是否找到单词
    char  en[MAX_CHAR];   //英文形式
    while(flag||choose)
    {
        printf("\n请输入你要修改的单词英文形式:");    
        scanf("%s",en);
        for(i=0;i<num;i++)//先找到该英文形式对应的序号
            if(strcmp(s[i].en,en)==0)
            {
                printf("\n请输入单词正确的英文形式:");
                scanf("%s",s[i].en);

                printf("\n请输入此单词正确的的中文形式:");
                scanf("%s",s[i].ch);

                printf("\n继续修改请选1，返回上一级请选0:");
                scanf("%d",&choose);
                if(choose==0) 
                return;    
            }
        flag=0;//找不到单词时，将flag变为0 
    }
    //找不到单词的处理 
   if(!flag)     
   printf("\n没有这个单词!");
}

//单词预览
void show()
{
    int  i=0;
    printf("\n单词：英文   中文 ");
    //循环输出单词 
    for(i=0;i<num;i++)
    printf("\n%-12s%-12s",s[i].en,s[i].ch);
}

//查询单词
void search()
{
    int i=0,choose=0,flag=1;
    char  ch[MAX_CHAR];   //中文形式
    while(choose||flag)
    {
        printf("\n请输入你要查询的单词中文形式:");    
        scanf("%s",ch);
        for(i=0;i<num;i++)
        //先找到该中文形式对应的序号
        {
            if(strcmp(s[i].ch,ch)==0)
            {        
                printf("\n英文形式       中文形式");
                printf("\n%-12s%12s",s[i].en,s[i].ch);
            }
       }
           printf("\n继续查询请选1，返回上一级请选0:");
        scanf("%d",&choose);
        if(choose==0) 
        return;    
        flag=0;
    }
    //没有单词的处理 
   if(!flag)     
   printf("\n没有这个单词!");
}

//中译英测试
void zytest()
{    
    int ifwrong=0;//1表示错误过，0表示一遍即正确测试通过 
    char b1[20];//输入英语单词答案的数组 
    int z;
    int choose=1;
    int tip;//作为复习错误单词，提示还是再输的标志 
    int  i;
    while(choose)
    { 
        i = rand()%num;//调用随机函数，随机取中文词 
        printf("\n【%s】请输入该词的英文形式:",s[i].ch);//针对显示随机选择的中文词 
        scanf("%s",&b1);//对于选择的单词的中文意思，写自己的答案 
        //输入错误处理 
        for(z=0;strcmp(b1,s[i].en)!=0;z=z)
        {
        printf("输入错误！！请重新输入:");
        scanf("%s",&b1);
        ifwrong=1;
        }
        //输入错误计分处理 
        if(ifwrong)
        {
            c++;
            ifwrong=0;
        }     
        printf("\n恭喜你，回答正确，加10分!\n\n");
        printf("\n进行错误单词复习请选2，继续测试请选1，返回上一级请选0:");
        scanf("%d",&choose);
        if(choose==0||choose==2) 
        {
            if(choose==2)//选择了单词复习 
            {
                printf("\n【%s】请输入该词的英文形式:",s[i].ch);//针对显示随机选择的中文词 
                scanf("%s",b1);//对于选择的单词的中文意思，写自己的答案
                //再次写错 
                if(strcmp(b1,s[i].en)!=0)
                {
                    printf("\n又错了/(ㄒoㄒ)/~~\n不扣分了，告诉你吧\n正确的形式是【%s】\n同学要好好背单词呀",s[i].en);
                }
                if(strcmp(b1,s[i].en)==0)//答对了
                {
                    printf("\n第一次为什么没有答对呢");                 
                 } 
            }
            if(choose==0)//返回上一级            
            return;
        }
        zytest();        
    }
}
